cy_en_adcmic_status_t Cy_ADCMic_SelectSource ( MXS40ADCMIC_Type *  base,
cy_en_adcmic_source_t  source 
)

Selects the specified ADC source.

Parameters
baseThe pointer to the hardware ADCMic block.
sourceThe source value cy_en_adcmic_source_t.
Note
When the source is switched, the sample rate and the DC conversion time (in case of switching to DC) are also reset accordingly:
  • if switch to CY_ADCMIC_DC the sample rate is reset to CY_ADCMIC_480KSPS and the DC conversion time is reset to CY_ADCMIC_DC_CONV_TIME_10_US.
  • if switch to CY_ADCMIC_MIC or CY_ADCMIC_PDM the sample rate is reset to CY_ADCMIC_16KSPS.
Returns
The status cy_en_adcmic_status_t.
Function Usage
/* Scenario: Presume the block is initialized before, now there is need to switch to the DC measurement: */
if (CY_ADCMIC_SUCCESS != Cy_ADCMic_SelectSource(MXS40ADCMIC0, CY_ADCMIC_DC))
{
/* Something went wrong, insert error handling here */
}
if (CY_ADCMIC_SUCCESS != Cy_ADCMic_SetDcConvTime(MXS40ADCMIC0, CY_ADCMIC_DC_CONV_TIME_20_US))
{
/* Something went wrong, insert error handling here */
}

__STATIC_INLINE void Cy_ADCMic_StartConvert ( MXS40ADCMIC_Type *  base)

Starts the conversion process.

Parameters
baseThe pointer to the hardware ADCMic block.
Function Usage
/* Scenario: Initialize ADCMic for the DC measurement: */
uint16_t data = 0U;
if (CY_ADCMIC_SUCCESS != Cy_ADCMic_Init(MXS40ADCMIC0, &ADCMicDcCfg))
{
/* Something went wrong, insert error handling here */
}
if (CY_ADCMIC_SUCCESS != Cy_ADCMic_SetDcConvTime(MXS40ADCMIC0, CY_ADCMIC_DC_CONV_TIME_15_US))
{
/* Something went wrong, insert error handling here */
}
Cy_ADCMic_SelectDcChannel(MXS40ADCMIC0, CY_ADCMIC_GPIO0);
Cy_ADCMic_Enable(MXS40ADCMIC0); /* Enable the block */
Cy_ADCMic_StartConvert(MXS40ADCMIC0); /* Start the conversion */
Cy_ADCMic_IsEndConversion(MXS40ADCMIC0, CY_ADCMIC_WAIT_FOR_RESULT); /* Wait the conversion is done */
Cy_ADCMic_StopConvert(MXS40ADCMIC0); /* Stop the conversion */
data = Cy_ADCMic_GetDcResult(MXS40ADCMIC0); /* Get the result */
